from collections import deque
class Solution(object):
    def findMaxValueOfEquation(self, points, k):
        """
        :type points: List[List[int]]
        :type k: int
        :rtype: int
        """
        dq = deque()  # 存 (yi - xi, xi)
        res = float('-inf')

        for x, y in points:
            while dq and x - dq[0][1] > k:
                dq.popleft()

            if dq:
                res = max(res, dq[0][0] + x + y)

            while dq and dq[-1][0] <= y - x:
                dq.pop()

            dq.append((y - x, x))

        return res
