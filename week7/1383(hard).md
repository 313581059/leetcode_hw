import heapq
class Solution(object):
    def maxPerformance(self, n, speed, efficiency, k):
        """
        :type n: int
        :type speed: List[int]
        :type efficiency: List[int]
        :type k: int
        :rtype: int
        """
        MOD = 10**9 + 7
        
        # (efficiency, speed)
        people = sorted(zip(efficiency, speed), reverse=True)
        
        heap = []
        speed_sum = 0
        ans = 0
        
        for e, s in people:
            heapq.heappush(heap, s)
            speed_sum += s
            
            if len(heap) > k:
                speed_sum -= heapq.heappop(heap)
            
            ans = max(ans, speed_sum * e)
        
        return ans % MOD
