class Solution(object):
    def maximumSumSubsequence(self, nums, queries):
        """
        :type nums: List[int]
        :type queries: List[List[int]]
        :rtype: int
        """

        n = len(nums)
        tree = [[0,0,0,0] for _ in range(4*n)]
        
        def build(idx, l, r):
            if l == r:
                v = max(0, nums[l])
                tree[idx] = [0, 0, 0, v]
                return
            
            mid = (l+r)//2
            build(idx*2+1, l, mid)
            build(idx*2+2, mid+1, r)
            tree[idx] = merge(tree[idx*2+1], tree[idx*2+2])
        
        def merge(L, R):
            res = [0]*4
            
            res[0] = max(L[0]+R[0], L[1]+R[0], L[0]+R[2])
            res[1] = max(L[0]+R[1], L[1]+R[1], L[0]+R[3])
            res[2] = max(L[2]+R[0], L[3]+R[0], L[2]+R[2])
            res[3] = max(L[2]+R[1], L[3]+R[1], L[2]+R[3])
            
            return res
        
        def update(idx, l, r, pos, val):
            if l == r:
                tree[idx] = [0,0,0,max(0,val)]
                return
            
            mid = (l+r)//2
            if pos <= mid:
                update(idx*2+1, l, mid, pos, val)
            else:
                update(idx*2+2, mid+1, r, pos, val)
            
            tree[idx] = merge(tree[idx*2+1], tree[idx*2+2])
        
        build(0,0,n-1)
        
        MOD = 10**9 + 7
        ans = 0
        
        for pos, val in queries:
            update(0,0,n-1,pos,val)
            ans = (ans + max(tree[0])) % MOD
        
        return ans
